// 查看系统主题偏好
console.log('System prefers dark:',
  window.matchMedia('(prefers-color-scheme: dark)').matches);

// 查看当前主题
console.log('Current theme:',
  document.body.classList.contains('dark-mode') ? 'dark' : 'light');
